    <script type="module">
        import { createApp } from 'https://unpkg.com/vue@3/dist/vue.esm-browser.js';
        
        // Firebase v11.6.1 SDK imports
        import { initializeApp } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js';
        import { getAuth, signInWithCustomToken, signInAnonymously, onAuthStateChanged } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js';
        import { getFirestore, doc, setDoc, getDoc, collection, onSnapshot, query, deleteDoc } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js';

        // Firebase Setup Logic
        let db, auth, appId, isCloud = false;
        try {
            if (typeof __firebase_config !== 'undefined' && __firebase_config) {
                const firebaseConfig = JSON.parse(__firebase_config);
                const app = initializeApp(firebaseConfig);
                auth = getAuth(app);
                db = getFirestore(app);
                appId = typeof __app_id !== 'undefined' ? __app_id : 'sagam-app-id';
                isCloud = true;
            }
        } catch(e) { 
            console.warn('Firebase config not found. Running in Local Storage mode.', e); 
        }

        createApp({
            data() {
                return {
                    isCloud: isCloud,
                    currentUser: null,
                    view: 'login', // 'login', 'admin', 'teacher'
                    showAdminLogin: false,
                    adminPwd: '',
                    
                    // DB State
                    rules: [], // 교사들의 제약조건 DB
                    
                    // Admin State
                    loading: false,
                    fileName: '',
                    workbook: null,
                    worksheet: null,
                    dataAOA: [],
                    colMap: {},
                    headerRowIdx: -1,
                    excelStaffMap: {}, // 엑셀에서 뽑아낸 각 교사별 행 위치 정보 저장소
                    stats: {},
                    scheduleReady: false,
                    previewSchedule: [],
                    
                    // Teacher State
                    selectedTeacherName: '',
                    teacherForm: null,
                }
            },
            computed: {
                statsArray() {
                    return Object.keys(this.stats).map(name => ({
                        name: name,
                        group: this.rules.find(r => r.name === name)?.group || '우정학사',
                        total: this.stats[name].total,
                        weekend: this.stats[name].weekend
                    })).sort((a, b) => b.total - a.total);
                }
            },
            async mounted() {
                // Initialize Auth & Data Sync
                if (this.isCloud) {
                    try {
                        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                            await signInWithCustomToken(auth, __initial_auth_token);
                        } else {
                            await signInAnonymously(auth);
                        }
                        
                        onAuthStateChanged(auth, (user) => {
                            this.currentUser = user;
                            if (user) this.listenToRules();
                        });
                    } catch(e) {
                        console.error("Auth Error", e);
                    }
                } else {
                    // Local Storage Fallback
                    const localRules = localStorage.getItem('sagam_rules');
                    if (localRules) this.rules = JSON.parse(localRules);
                }
            },
            methods: {
                // =============== 화면 이동 & 로그인 ===============
                loginAdmin() {
                    if (this.adminPwd === '0000') {
                        this.view = 'admin';
                        this.adminPwd = '';
                    } else {
                        alert('비밀번호가 틀렸습니다. (기본값: 0000)');
                    }
                },
                
                // =============== 데이터 동기화 (Firestore / LocalStorage) ===============
                listenToRules() {
                    if (!this.currentUser || !this.isCloud) return;
                    const q = query(collection(db, 'artifacts', appId, 'public', 'data', 'sagamRules'));
                    onSnapshot(q, (snapshot) => {
                        const loaded = [];
                        snapshot.forEach((doc) => {
                            loaded.push({ name: doc.id, ...doc.data() });
                        });
                        this.rules = loaded;
                    }, (error) => {
                        console.error("Snapshot error:", error);
                    });
                },
                syncData() {
                    // 수동 새로고침 (로컬 환경 대응)
                    if (!this.isCloud) {
                        const localRules = localStorage.getItem('sagam_rules');
                        if (localRules) this.rules = JSON.parse(localRules);
                    }
                },
                async saveRule(rule) {
                    if (this.isCloud && this.currentUser) {
                        const docRef = doc(db, 'artifacts', appId, 'public', 'data', 'sagamRules', rule.name);
                        await setDoc(docRef, rule);
                    } else {
                        const idx = this.rules.findIndex(r => r.name === rule.name);
                        if (idx > -1) this.rules[idx] = rule;
                        else this.rules.push(rule);
                        localStorage.setItem('sagam_rules', JSON.stringify(this.rules));
                    }
                },
                async deleteRule(name) {
                    if(!confirm(`정말 ${name} 교사를 명단에서 삭제하시겠습니까?`)) return;
                    
                    if (this.isCloud && this.currentUser) {
                        await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'sagamRules', name));
                    } else {
                        this.rules = this.rules.filter(r => r.name !== name);
                        localStorage.setItem('sagam_rules', JSON.stringify(this.rules));
                    }
                },

                // =============== 교사 모드 로직 ===============
                loadTeacherRule() {
                    if (!this.selectedTeacherName) {
                        this.teacherForm = null;
                        return;
                    }
                    const existing = this.rules.find(r => r.name === this.selectedTeacherName);
                    this.teacherForm = JSON.parse(JSON.stringify(existing)); // 깊은 복사
                },
                async saveTeacherForm() {
                    await this.saveRule(this.teacherForm);
                    alert(`${this.teacherForm.name} 선생님의 설정이 저장되었습니다!`);
                },

                // =============== 관리자 모드: 엑셀 파싱 ===============
                async handleFileUpload(event) {
                    const file = event.target.files[0];
                    if (!file) return;
                    
                    this.loading = true;
                    this.fileName = file.name;
                    this.scheduleReady = false;

                    try {
                        const arrayBuffer = await file.arrayBuffer();
                        this.workbook = new ExcelJS.Workbook();
                        await this.workbook.xlsx.load(arrayBuffer);
                        
                        this.worksheet = this.workbook.worksheets[0];
                        this.dataAOA = [];

                        this.worksheet.eachRow({ includeEmpty: true }, (row, rowNumber) => {
                            let rowValues = [];
                            row.eachCell({ includeEmpty: true }, (cell, colNumber) => {
                                let val = cell.value;
                                if (val && typeof val === 'object') {
                                    if (val.result !== undefined) val = val.result;
                                    else if (val.richText) val = val.richText.map(t => t.text).join('');
                                }
                                rowValues[colNumber - 1] = val === undefined || val === null ? "" : String(val);
                            });
                            this.dataAOA[rowNumber - 1] = rowValues;
                        });

                        this.analyzeStructure();
                    } catch (error) {
                        alert("파일을 읽는 중 오류가 발생했습니다.");
                        console.error(error);
                    } finally {
                        this.loading = false;
                    }
                },

                analyzeStructure() {
                    this.colMap = { date: -1, day: -1, shift: -1, ujeong: -1, bonghwang: -1, name: -1, total: -1, groupCol: -1 };
                    let headerRowIdx = -1;

                    for (let i = 0; i < Math.min(20, this.dataAOA.length); i++) {
                        const row = this.dataAOA[i];
                        if (!row) continue;
                        const rowStrs = row.map(c => String(c || '').replace(/\s+/g, ''));
                        
                        if (rowStrs.includes('일자') && rowStrs.includes('성명')) {
                            headerRowIdx = i;
                            this.colMap.date = rowStrs.indexOf('일자');
                            this.colMap.day = rowStrs.indexOf('요일');
                            this.colMap.shift = rowStrs.findIndex(s => s.includes('주야') || s.includes('주·야간'));
                            this.colMap.ujeong = rowStrs.findIndex(s => s.includes('우정학사'));
                            this.colMap.bonghwang = rowStrs.findIndex(s => s.includes('봉황관') || s === '봉황');
                            this.colMap.name = rowStrs.indexOf('성명');
                            this.colMap.total = rowStrs.findIndex(s => s.includes('누계') || s.includes('시수'));
                            this.colMap.groupCol = rowStrs.findIndex(s => s === '근무' || s === '소속');
                            break;
                        }
                    }

                    this.headerRowIdx = headerRowIdx;
                    if (headerRowIdx === -1) {
                        alert("엑셀 양식을 인식할 수 없습니다.");
                        return;
                    }

                    if (this.colMap.bonghwang === -1 && this.colMap.ujeong !== -1) {
                        this.colMap.bonghwang = this.colMap.ujeong + 1; 
                    }

                    let lastGroup = '우정학사';
                    this.excelStaffMap = {}; // 초기화

                    for (let i = headerRowIdx + 1; i < this.dataAOA.length; i++) {
                        const row = this.dataAOA[i];
                        if (!row) continue;

                        const nameVal = row[this.colMap.name];
                        const groupVal = this.colMap.groupCol !== -1 ? row[this.colMap.groupCol] : '';

                        if (groupVal && groupVal.trim() !== "") {
                            if (groupVal.includes('봉황')) lastGroup = '봉황관';
                            else if (groupVal.includes('우정')) lastGroup = '우정학사';
                        }

                        if (nameVal && nameVal.trim() !== "") {
                            const cleanName = nameVal.trim();
                            const invalidNames = ['성명', '이름', '연락처', '전화번호', '누계', '시수', '비고', '소속', '근무', '우정학사', '봉황관', '주간', '야간', '일자', '요일', '주·야간'];
                            if (invalidNames.includes(cleanName.replace(/\s+/g, ''))) continue;
                            if (/^[0-9-]+$/.test(cleanName)) continue;

                            // 엑셀 행 매핑 기록 (누계 작성 시 사용)
                            this.excelStaffMap[cleanName] = { rowIdx: i };

                            // DB에 없는 교사면 자동 생성하여 업로드
                            if (!this.rules.find(r => r.name === cleanName)) {
                                const newRule = {
                                    name: cleanName,
                                    group: lastGroup,
                                    startDate: 1,
                                    excludedDays: [],
                                    fixedDays: [],
                                    excludedDatesStr: ""
                                };
                                this.saveRule(newRule);
                            }
                        }
                    }
                    alert("엑셀 구조 파싱 및 클라우드 명단 동기화 완료!\n이제 교사들이 링크로 접속해 조건을 입력할 수 있습니다.");
                },

                // =============== 관리자 모드: 알고리즘 실행 ===============
                runAlgorithm() {
                    if(!this.dataAOA || this.rules.length === 0) return;

                    this.stats = {};
                    this.rules.forEach(rule => {
                        this.stats[rule.name] = { total: 0, weekend: 0, lastWorkedDateNum: -99 };
                    });

                    const getDateNum = (val) => {
                        if (!val) return -1;
                        const match = String(val).match(/\d+/);
                        return match ? parseInt(match[0]) : -1;
                    };

                    const { date: cDate, day: cDay, shift: cShift, ujeong: cUjeong, bonghwang: cBonghwang } = this.colMap;

                    for (let i = 0; i < this.dataAOA.length; i++) {
                        const row = this.dataAOA[i];
                        if (!row) continue;
                        const dateNum = getDateNum(row[cDate]);
                        if (dateNum === -1) continue;

                        const isWeekend = (String(row[cDay] || '').trim() === '토' || String(row[cDay] || '').trim() === '일');

                        const processExisting = (assignedName) => {
                            if (assignedName && this.stats[assignedName]) {
                                this.stats[assignedName].total += 1;
                                this.stats[assignedName].lastWorkedDateNum = dateNum;
                                if (isWeekend) this.stats[assignedName].weekend += 1;
                            }
                        };
                        if (cUjeong !== -1) processExisting(row[cUjeong]);
                        if (cBonghwang !== -1) processExisting(row[cBonghwang]);
                    }

                    this.previewSchedule = [];

                    const getCandidate = (targetGroup, dateNum, dayStr, isWeekend) => {
                        let candidates = this.rules.filter(rule => {
                            if (rule.group !== targetGroup) return false;
                            
                            const stat = this.stats[rule.name];
                            if (dateNum < rule.startDate) return false;
                            
                            const parsedExDates = rule.excludedDatesStr.split(',').map(s => parseInt(s.trim())).filter(n => !isNaN(n));
                            if (parsedExDates.includes(dateNum)) return false;
                            if (rule.excludedDays.includes(dayStr)) return false;
                            
                            if (!isWeekend && rule.fixedDays.length > 0) {
                                if (!rule.fixedDays.includes(dayStr)) return false;
                            }
                            
                            if (dateNum === stat.lastWorkedDateNum || dateNum === stat.lastWorkedDateNum + 1) return false;
                            return true;
                        });

                        if (candidates.length === 0) return null;

                        candidates.sort(() => Math.random() - 0.5);
                        candidates.sort((a, b) => {
                            if (isWeekend) {
                                if(this.stats[a.name].weekend !== this.stats[b.name].weekend) return this.stats[a.name].weekend - this.stats[b.name].weekend;
                            } else {
                                const aWk = this.stats[a.name].total - this.stats[a.name].weekend;
                                const bWk = this.stats[b.name].total - this.stats[b.name].weekend;
                                if(aWk !== bWk) return aWk - bWk;
                            }
                            return this.stats[a.name].total - this.stats[b.name].total;
                        });
                        return candidates[0].name;
                    };

                    for (let i = 0; i < this.dataAOA.length; i++) {
                        const row = this.dataAOA[i];
                        if (!row) continue;
                        const dateNum = getDateNum(row[cDate]);
                        if (dateNum === -1) continue;

                        const dayStr = String(row[cDay] || '').trim();
                        const shiftStr = String(row[cShift] || '').trim();
                        const isWeekend = (dayStr === '토' || dayStr === '일');

                        let ujeongAssignee = row[cUjeong] || '';
                        let bonghwangAssignee = row[cBonghwang] || '';

                        if (!ujeongAssignee && cUjeong !== -1) {
                            const chosen = getCandidate('우정학사', dateNum, dayStr, isWeekend);
                            if (chosen) {
                                ujeongAssignee = chosen;
                                row[cUjeong] = chosen;
                                row._modified_ujeong = true;
                                this.stats[chosen].total += 1;
                                this.stats[chosen].lastWorkedDateNum = dateNum;
                                if (isWeekend) this.stats[chosen].weekend += 1;
                            }
                        }

                        if (!bonghwangAssignee && cBonghwang !== -1) {
                            const chosen = getCandidate('봉황관', dateNum, dayStr, isWeekend);
                            if (chosen) {
                                bonghwangAssignee = chosen;
                                row[cBonghwang] = chosen;
                                row._modified_bonghwang = true;
                                this.stats[chosen].total += 1;
                                this.stats[chosen].lastWorkedDateNum = dateNum;
                                if (isWeekend) this.stats[chosen].weekend += 1;
                            }
                        }

                        this.previewSchedule.push({ 
                            date: row[cDate], day: dayStr, shift: shiftStr, 
                            ujeongAssignee: ujeongAssignee, bonghwangAssignee: bonghwangAssignee 
                        });
                    }
                    this.scheduleReady = true;
                    setTimeout(() => { window.scrollTo({ top: document.body.scrollHeight, behavior: 'smooth' }); }, 100);
                },

                // =============== 엑셀 다운로드 (기존 로직 유지) ===============
                async downloadExcel() {
                    const { ujeong: cUjeong, bonghwang: cBonghwang, total: cTotal } = this.colMap;

                    for (let r = 0; r < this.dataAOA.length; r++) {
                        const rowData = this.dataAOA[r];
                        if (!rowData) continue;

                        const excelRow = r + 1;
                        if (rowData._modified_ujeong && cUjeong !== -1) this.worksheet.getCell(excelRow, cUjeong + 1).value = rowData[cUjeong];
                        if (rowData._modified_bonghwang && cBonghwang !== -1) this.worksheet.getCell(excelRow, cBonghwang + 1).value = rowData[cBonghwang];
                    }

                    if (cTotal !== -1) {
                        for (let rule of this.rules) {
                            // 현재 업로드된 엑셀 상의 행 위치 조회
                            const mapping = this.excelStaffMap[rule.name];
                            if (mapping && mapping.rowIdx !== undefined && this.stats[rule.name]) {
                                const excelRow = mapping.rowIdx + 1;
                                this.worksheet.getCell(excelRow, cTotal + 1).value = this.stats[rule.name].total;
                                
                                const weekendCnt = this.stats[rule.name].weekend;
                                const weekdayCnt = this.stats[rule.name].total - weekendCnt;
                                this.worksheet.getCell(excelRow, 14).value = weekendCnt;
                                this.worksheet.getCell(excelRow, 15).value = weekdayCnt;
                            }
                        }
                    }

                    const colQ = 17;
                    const q4Cell = this.worksheet.getCell(4, colQ);
                    q4Cell.value = "🔍 사감 검색";
                    q4Cell.font = { bold: true };
                    q4Cell.alignment = { horizontal: 'center' };
                    
                    const q5Cell = this.worksheet.getCell(5, colQ);
                    const staffNames = this.rules.map(r => r.name).join(',');
                    q5Cell.dataValidation = { type: 'list', allowBlank: true, formulae: [`"${staffNames}"`] };
                    q5Cell.value = "이름 선택 ▼";
                    q5Cell.fill = { type: 'pattern', pattern: 'solid', fgColor: { argb: 'FFF2F2F2' } };
                    q5Cell.border = { top: {style:'thin'}, left: {style:'thin'}, bottom: {style:'thin'}, right: {style:'thin'} };

                    if (this.headerRowIdx !== -1 && cUjeong !== -1) {
                        const startRow = this.headerRowIdx + 2;
                        const endRow = this.dataAOA.length;
                        
                        const getColLetter = (idx) => {
                            let temp, letter = '';
                            let col = idx + 1;
                            while (col > 0) {
                                temp = (col - 1) % 26;
                                letter = String.fromCharCode(temp + 65) + letter;
                                col = (col - temp - 1) / 26;
                            }
                            return letter;
                        };

                        const ujeongLetter = getColLetter(cUjeong);
                        const bonghwangLetter = cBonghwang !== -1 ? getColLetter(cBonghwang) : ujeongLetter;

                        this.worksheet.addConditionalFormatting({
                            ref: `${ujeongLetter}${startRow}:${bonghwangLetter}${endRow}`,
                            rules: [{
                                type: 'cellIs', operator: 'equal', formulae: ['$Q$5'],
                                style: {
                                    fill: { type: 'pattern', pattern: 'solid', bgColor: { argb: 'FFFFFF00' } },
                                    font: { color: { argb: 'FF0000FF' }, bold: true }
                                }
                            }]
                        });
                    }

                    const buffer = await this.workbook.xlsx.writeBuffer();
                    const blob = new Blob([buffer], { type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet' });
                    const url = window.URL.createObjectURL(blob);
                    
                    const a = document.createElement('a');
                    a.href = url;
                    a.download = `${this.fileName.replace(/\.[^/.]+$/, "")}_자동배정완료.xlsx`;
                    a.click();
                    
                    window.URL.revokeObjectURL(url);
                }
            }
        }).mount('#app')
    </script>
